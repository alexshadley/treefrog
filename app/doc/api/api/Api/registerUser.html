registerUser method

Future<RegistrationResult> registerUser (String email, String displayName, String method, [ String password ])

Creates a user in the backend database. password should only be used if the user is being created using the Email sign-in method. This will return the appropriate RegistrationResult for the situation encountered.

Implementation 

Future<RegistrationResult> registerUser(String email, String displayName, String method, [String password]) async {
  if (!_ready) {
    await _config.init();
    _ready = true;
  }

  var body;

  if (password != null && password.isNotEmpty) {
    password = util.hash(password);
    body = { 'email': email, 'display_name': displayName, 'password_hash': password, 'method': method };
  }
  else {
    body = { 'email': email, 'display_name': displayName, 'method': method };
  }

  var response = await http.post("${_config.getValue("api_url")}/users/", body: body);

  if (response.statusCode == 201)
    return RegistrationResult.SUCCESS;
  else if(response.statusCode == 400)
    return RegistrationResult.DUPLICATE_EMAIL;
  else
    return RegistrationResult.FAILURE;
}
